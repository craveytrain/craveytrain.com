Module pattern

  • Private vars/functions
  • Runs when loaded and exposes public methods/vars
(function(context) {
  var privateVar = 'keep it on the dl',
      plugin = {
        publicMethod: function() {
          return privateVar;
        }
      };

  context.plugin = plugin;
}(namespace));

plugin.privateVar // undefined
plugin.publicMethod() // 'keep it on the dl'

jQuery Collections

  • Think of it as an array of DOM elements with custom methods
  • 0 indexed
var $links = $('a') // jQuery Collection
var link = $('a')[0] // DOM Element

$links.addClass('some') // adds class to the links
link.addClass('some') // fails, not a jQ collection

$links.href // fails, not a DOM node
link.href // returns the href attribute value
$links[0].href // returns the href attribute
$links.eq(0).attr('href') // jQ method (slower)

Event Trigger

trigger

  • Use this over helper (.click())
  • Simulates a real event (trigger handlers and bubble up DOM)
  • Does not perform default behavior
$('.some').trigger('click');

triggerHandler

  • Does not bubble up DOM
  • Executes the handler for first element of the jQ collection
$('.some').triggerHandler('click');
$.triggerHandler()

Event Delegation

Delegation puts a listener further up the DOM and uses bubbling to see if the selected element was the target

// Non-delegated event listeners
$('a').on('click', function() {...}); // 500 links = 500 event listeners

// Delegated event listeners
$(document).on('click', 'a', function() {...}); // Delegated, 1 event listener

<document>      ↑
  <body>        ↑
    <a>      (click)

$(document).off('click', 'a'); // Undelegate

inArray

  • Returns the index of something if it's in an array
  • Otherwise it returns -1
  • jQuery version of Array.indexOf()
var arr = [2, 3, 4, 5];
$.inArray(2, arr) // returns 0
$.inArray(7, arr) // returns -1
~$.inArray(7, arr) // return 0 (if you don't need index, just testing if is in array)

Extend

  • Merge Objects into each other
  • Create a copy of an object (or merged objects)
  • true makes it a deep (recursive) copy
// merge newObj into origObj
$.extend(origObj, newObj)

// newObj is a copy of origObj
var newObj = $.extend({}, origObj)

// recursively merges newObj into origObj
$.extend(true, origObj, newObj)

  • success is defined by a 200 response (400 & 500 are errors)

$.getJSON('/url', {reqParam: value}, function(d) {
  ... // success callback, d = parsed json
});

$.post('/url', $form.serialize(), function(d) {
  ... // success callback, d = response
});

Notable Ajax Methods

$.ajax({
  url: '/url',
  type: 'get', // method
  data: { request: param }, // or post payload
  dataType: 'json', // type of response data
  success: function(){},
  error: function() {}, // (400, 500)
  beforeSend: function() {
    $throbber.show();
  },
  complete: function() { // success and error

  }
